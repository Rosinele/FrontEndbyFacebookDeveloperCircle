importar a biblioteca express
arquivo.js: var expressjs = require("express");
 
criar primeira aplicação que sera iniciada atraves da biblioteca express
arquivo.js: var aplicacao = expressjs;
 
Disponibilizar porta para acesso à aplicação
arquivo.js: aplicacao.listen(8989);

Adicionar express ao projeto
terminal node: npm install express

executar projeto
node index.js
